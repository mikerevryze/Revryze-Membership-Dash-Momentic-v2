Replace the entire contents of `frontend/index.html` with exactly the following code. Do not modify any other files — `main.py` and all other files stay exactly as-is.

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0" />
  <title>Revryze — Pre-Launch Dashboard</title>
  <script src="https://cdn.tailwindcss.com"></script>
  <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
  <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
  <script src="https://unpkg.com/prop-types@15/prop-types.min.js"></script>
  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  <script src="https://unpkg.com/recharts@2.5.0/umd/Recharts.js"></script>
  <style>
    body { background: #0f1117; color: #e2e8f0; font-family: 'Inter', system-ui, sans-serif; }
    input[type="date"]::-webkit-calendar-picker-indicator { filter: invert(1); }
    .kpi-card { background: #1a1d2e; border: 1px solid #2a2d3e; }
    .tab-active { background: #3b82f6; color: #fff; }
    .tab-inactive { background: #1a1d2e; color: #94a3b8; border: 1px solid #2a2d3e; }
    .chart-container { background: #1a1d2e; border: 1px solid #2a2d3e; }
    select, input[type="date"], input[type="number"] {
      background: #1a1d2e; color: #e2e8f0; border: 1px solid #2a2d3e;
      padding: 6px 10px; border-radius: 6px; outline: none;
    }
    select:focus, input:focus { border-color: #3b82f6; }
  </style>
</head>
<body>
  <div id="root"></div>
  <script type="text/babel">
    const { useState, useEffect, useCallback } = React;
    const {
      LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer, Legend
    } = Recharts;

    function formatCurrency(n) {
      if (n == null) return "$0.00";
      return "$" + Number(n).toLocaleString("en-US", { minimumFractionDigits: 2, maximumFractionDigits: 2 });
    }
    function formatNumber(n) {
      if (n == null) return "0";
      return Number(n).toLocaleString("en-US");
    }

    function KPICard({ label, value, prefix }) {
      return (
        <div className="kpi-card rounded-xl p-5 flex flex-col justify-between min-h-[110px]">
          <div className="text-sm text-slate-400 font-medium uppercase tracking-wide">{label}</div>
          <div className="text-2xl lg:text-3xl font-bold text-white mt-2">
            {prefix}{value}
          </div>
        </div>
      );
    }

    function App() {
      const [locations, setLocations] = useState([]);
      const [activeLocation, setActiveLocation] = useState("");
      const [summary, setSummary] = useState(null);
      const [daily, setDaily] = useState([]);
      const [config, setConfig] = useState(null);
      const [dateRange, setDateRange] = useState("all");
      const [customStart, setCustomStart] = useState("");
      const [customEnd, setCustomEnd] = useState("");
      const [loading, setLoading] = useState(true);
      const [error, setError] = useState(null);
      const [settingsOpen, setSettingsOpen] = useState(false);

      const getDateParams = useCallback(() => {
        const today = new Date();
        let start = null, end = today.toISOString().split("T")[0];
        if (dateRange === "7") {
          const d = new Date(today); d.setDate(d.getDate() - 7);
          start = d.toISOString().split("T")[0];
        } else if (dateRange === "30") {
          const d = new Date(today); d.setDate(d.getDate() - 30);
          start = d.toISOString().split("T")[0];
        } else if (dateRange === "60") {
          const d = new Date(today); d.setDate(d.getDate() - 60);
          start = d.toISOString().split("T")[0];
        } else if (dateRange === "custom") {
          start = customStart || null;
          end = customEnd || end;
        } else {
          start = null; end = null;
        }
        return { start, end };
      }, [dateRange, customStart, customEnd]);

      const fetchData = useCallback(async (loc) => {
        if (!loc) return;
        setLoading(true);
        setError(null);
        try {
          const { start, end } = getDateParams();
          let summaryUrl = `/api/summary?location=${encodeURIComponent(loc)}`;
          let dailyUrl = `/api/daily?location=${encodeURIComponent(loc)}`;
          if (start) { summaryUrl += `&start_date=${start}`; dailyUrl += `&start_date=${start}`; }
          if (end) { summaryUrl += `&end_date=${end}`; dailyUrl += `&end_date=${end}`; }

          const [sRes, dRes] = await Promise.all([fetch(summaryUrl), fetch(dailyUrl)]);
          if (!sRes.ok) throw new Error("Failed to fetch summary");
          if (!dRes.ok) throw new Error("Failed to fetch daily data");
          setSummary(await sRes.json());
          setDaily(await dRes.json());
        } catch (e) {
          setError(e.message);
        } finally {
          setLoading(false);
        }
      }, [getDateParams]);

      useEffect(() => {
        (async () => {
          try {
            const [locRes, cfgRes] = await Promise.all([fetch("/api/locations"), fetch("/api/config")]);
            const locs = await locRes.json();
            const cfg = await cfgRes.json();
            setLocations(locs);
            setConfig(cfg);
            if (locs.length > 0) {
              setActiveLocation(locs[0]);
            }
          } catch (e) {
            setError("Failed to load initial data: " + e.message);
            setLoading(false);
          }
        })();
      }, []);

      useEffect(() => {
        if (activeLocation) fetchData(activeLocation);
      }, [activeLocation, fetchData]);

      const saveConfig = async (newCfg) => {
        setConfig(newCfg);
        await fetch("/api/config", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(newCfg),
        });
      };

      const updateLocationConfig = (field, value) => {
        if (!config) return;
        const newCfg = JSON.parse(JSON.stringify(config));
        if (!newCfg.locations[activeLocation]) {
          newCfg.locations[activeLocation] = { sales_start_date: null, opening_date: null };
        }
        newCfg.locations[activeLocation][field] = value || null;
        saveConfig(newCfg);
        if (field === "sales_start_date") {
          fetchData(activeLocation);
        }
      };

      const updateAttrition = (field, value) => {
        if (!config) return;
        const newCfg = JSON.parse(JSON.stringify(config));
        newCfg.attrition[field] = field === "mode" ? value : parseFloat(value) || 0;
        saveConfig(newCfg);
      };

      const locConfig = config?.locations?.[activeLocation] || {};
      const openingDate = locConfig.opening_date;
      const salesStartDate = locConfig.sales_start_date;

      const today = new Date();
      const daysUntilOpening = openingDate
        ? Math.max(0, Math.ceil((new Date(openingDate) - today) / (1000 * 60 * 60 * 24)))
        : null;
      const projectedTotal = summary && summary.memberships_per_day > 0 && daysUntilOpening != null
        ? Math.round(summary.memberships_sold + summary.memberships_per_day * daysUntilOpening)
        : null;

      const ltvMultiplier = config?.attrition?.mode === "attrition_rate"
        ? (config.attrition.attrition_rate > 0 ? 1 / (config.attrition.attrition_rate / 100) : 0)
        : (config?.attrition?.avg_monthly_stay || 0);
      const estimatedLTV = summary ? summary.total_membership_revenue * ltvMultiplier : 0;

      const cumulativeDaily = [];
      let cumSum = 0;
      for (let i = 0; i < daily.length; i++) {
        cumSum += (daily[i].memberships_sold || 0);
        cumulativeDaily.push({
          ...daily[i],
          report_date: daily[i].report_date,
          ad_spend: daily[i].ad_spend,
          cumulative_memberships: cumSum,
        });
      }

      let projectionData = [];
      if (cumulativeDaily.length > 0 && openingDate && summary?.memberships_per_day > 0) {
        const lastPoint = cumulativeDaily[cumulativeDaily.length - 1];
        const lastDate = lastPoint.report_date;
        const lastCum = lastPoint.cumulative_memberships;
        projectionData = [
          { report_date: lastDate, projected: lastCum },
          { report_date: openingDate, projected: projectedTotal },
        ];
      }

      const mergedChartData = (() => {
        const map = {};
        cumulativeDaily.forEach(d => {
          map[d.report_date] = { ...d };
        });
        projectionData.forEach(d => {
          if (map[d.report_date]) {
            map[d.report_date].projected = d.projected;
          } else {
            map[d.report_date] = { report_date: d.report_date, projected: d.projected };
          }
        });
        return Object.values(map).sort((a, b) => a.report_date.localeCompare(b.report_date));
      })();

      return (
        <div className="min-h-screen px-4 py-6 max-w-7xl mx-auto">
          {/* Header */}
          <div className="flex items-center justify-between mb-8">
            <div>
              <h1 className="text-3xl font-bold text-white tracking-tight">
                <span className="text-blue-500">Revryze</span> Pre-Launch Dashboard
              </h1>
            </div>
          </div>

          {/* Location Tabs */}
          <div className="flex gap-2 mb-6 flex-wrap">
            {locations.map((loc) => (
              <button
                key={loc}
                onClick={() => setActiveLocation(loc)}
                className={`px-5 py-2.5 rounded-lg font-semibold text-sm transition-all ${
                  activeLocation === loc ? "tab-active" : "tab-inactive hover:bg-slate-700"
                }`}
              >
                {loc}
              </button>
            ))}
          </div>

          {/* Date Range */}
          <div className="flex items-center gap-3 mb-6 flex-wrap">
            <select
              value={dateRange}
              onChange={(e) => setDateRange(e.target.value)}
              className="rounded-lg text-sm"
            >
              <option value="all">All Time</option>
              <option value="7">Last 7 Days</option>
              <option value="30">Last 30 Days</option>
              <option value="60">Last 60 Days</option>
              <option value="custom">Custom</option>
            </select>
            {dateRange === "custom" && (
              <>
                <input
                  type="date"
                  value={customStart}
                  onChange={(e) => setCustomStart(e.target.value)}
                  className="rounded-lg text-sm"
                />
                <span className="text-slate-400">to</span>
                <input
                  type="date"
                  value={customEnd}
                  onChange={(e) => setCustomEnd(e.target.value)}
                  className="rounded-lg text-sm"
                />
                <button
                  onClick={() => fetchData(activeLocation)}
                  className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-lg text-sm font-medium"
                >
                  Apply
                </button>
              </>
            )}
          </div>

          {error && (
            <div className="bg-red-900/50 border border-red-700 text-red-200 px-4 py-3 rounded-lg mb-6">
              {error}
            </div>
          )}

          {summary && summary.data_source === "sample" && (
            <div className="bg-amber-900/30 border border-amber-600/30 text-amber-300 px-4 py-2.5 rounded-lg mb-6 text-sm">
              Showing sample data — Snowflake is not connected. Data will update automatically once Snowflake is available.
            </div>
          )}

          {loading ? (
            <div className="text-center py-20 text-slate-400 text-lg">Loading data...</div>
          ) : summary ? (
            <>
              {/* KPI Cards */}
              <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-8">
                <KPICard label="Total Ad Spend" value={formatCurrency(summary.total_ad_spend)} />
                <KPICard label="Meta Leads" value={formatNumber(summary.total_meta_leads)} />
                <KPICard label="CPL" value={formatCurrency(summary.cpl)} />
                <KPICard label="Memberships Sold" value={formatNumber(summary.memberships_sold)} />
                <KPICard label="Total Revenue" value={formatCurrency(summary.total_membership_revenue)} />
                <KPICard label="Cost Per Member" value={summary.memberships_sold > 0 ? formatCurrency(summary.cost_per_membership) : "—"} />
                <KPICard label="Avg Membership Value" value={summary.memberships_sold > 0 ? formatCurrency(summary.total_membership_revenue / summary.memberships_sold) : "—"} />
              </div>

              {/* LTV Callout */}
              <div className="kpi-card rounded-xl p-6 mb-8">
                <div className="flex flex-col md:flex-row md:items-center gap-6">
                  <div className="flex-1">
                    <div className="text-sm text-slate-400 mb-2">Attrition Model</div>
                    <div className="flex gap-4 mb-3">
                      <label className="flex items-center gap-2 text-sm text-slate-300 cursor-pointer">
                        <input
                          type="radio"
                          name="attrition_mode_ltv"
                          checked={config?.attrition?.mode === "attrition_rate"}
                          onChange={() => updateAttrition("mode", "attrition_rate")}
                          className="accent-blue-500"
                        />
                        Monthly Attrition Rate (%)
                      </label>
                      <label className="flex items-center gap-2 text-sm text-slate-300 cursor-pointer">
                        <input
                          type="radio"
                          name="attrition_mode_ltv"
                          checked={config?.attrition?.mode === "avg_monthly_stay"}
                          onChange={() => updateAttrition("mode", "avg_monthly_stay")}
                          className="accent-blue-500"
                        />
                        Avg Monthly Stay (months)
                      </label>
                    </div>
                    {config?.attrition?.mode === "attrition_rate" ? (
                      <input
                        type="number"
                        step="0.1"
                        min="0"
                        value={config.attrition.attrition_rate}
                        onChange={(e) => updateAttrition("attrition_rate", e.target.value)}
                        className="w-48 rounded-lg"
                        placeholder="e.g. 5.0"
                      />
                    ) : (
                      <input
                        type="number"
                        step="1"
                        min="0"
                        value={config?.attrition?.avg_monthly_stay || 12}
                        onChange={(e) => updateAttrition("avg_monthly_stay", e.target.value)}
                        className="w-48 rounded-lg"
                        placeholder="e.g. 12"
                      />
                    )}
                  </div>
                  <div className="flex-1 bg-emerald-900/30 border border-emerald-700/30 rounded-lg p-5">
                    <div className="text-sm text-slate-400 mb-1">Estimated Lifetime Value Revryze Has Sold</div>
                    <div className="text-3xl font-bold text-emerald-400">{formatCurrency(estimatedLTV)}</div>
                    <div className="text-xs text-slate-500 mt-1">
                      Revenue ({formatCurrency(summary.total_membership_revenue)}) × LTV Multiplier ({ltvMultiplier.toFixed(1)} months)
                    </div>
                  </div>
                </div>
              </div>

              {/* Charts */}
              <div className="grid grid-cols-1 lg:grid-cols-2 gap-6 mb-8">
                <div className="chart-container rounded-xl p-5">
                  <h3 className="text-lg font-semibold text-white mb-4">Daily Ad Spend</h3>
                  <ResponsiveContainer width="100%" height={300}>
                    <LineChart data={daily}>
                      <CartesianGrid strokeDasharray="3 3" stroke="#2a2d3e" />
                      <XAxis dataKey="report_date" stroke="#64748b" tick={{ fontSize: 11 }} />
                      <YAxis stroke="#64748b" tick={{ fontSize: 11 }} />
                      <Tooltip
                        contentStyle={{ background: "#1a1d2e", border: "1px solid #2a2d3e", borderRadius: 8 }}
                        labelStyle={{ color: "#94a3b8" }}
                      />
                      <Line type="monotone" dataKey="ad_spend" stroke="#3b82f6" strokeWidth={2} dot={false} name="Ad Spend ($)" />
                    </LineChart>
                  </ResponsiveContainer>
                </div>

                <div className="chart-container rounded-xl p-5">
                  <h3 className="text-lg font-semibold text-white mb-4">Cumulative Memberships Sold</h3>
                  <ResponsiveContainer width="100%" height={300}>
                    <LineChart data={mergedChartData}>
                      <CartesianGrid strokeDasharray="3 3" stroke="#2a2d3e" />
                      <XAxis dataKey="report_date" stroke="#64748b" tick={{ fontSize: 11 }} />
                      <YAxis stroke="#64748b" tick={{ fontSize: 11 }} />
                      <Tooltip
                        contentStyle={{ background: "#1a1d2e", border: "1px solid #2a2d3e", borderRadius: 8 }}
                        labelStyle={{ color: "#94a3b8" }}
                      />
                      <Line type="monotone" dataKey="cumulative_memberships" stroke="#3b82f6" strokeWidth={2} dot={false} name="Actual" />
                      <Line type="monotone" dataKey="projected" stroke="#f59e0b" strokeWidth={2} strokeDasharray="6 3" dot={false} name="Projected" />
                      <Legend />
                    </LineChart>
                  </ResponsiveContainer>
                </div>
              </div>

              {/* Sales Velocity */}
              <div className="chart-container rounded-xl p-6 mb-8">
                <h3 className="text-lg font-semibold text-white mb-4">Sales Velocity</h3>
                <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
                  <div>
                    <label className="text-sm text-slate-400 block mb-1">Sales Start Date</label>
                    <input
                      type="date"
                      value={salesStartDate || ""}
                      onChange={(e) => updateLocationConfig("sales_start_date", e.target.value)}
                      className="w-full rounded-lg"
                    />
                  </div>
                  <div>
                    <div className="text-sm text-slate-400 mb-1">Current Velocity</div>
                    <div className="text-xl font-bold text-white">
                      {summary.memberships_per_day} <span className="text-sm text-slate-400 font-normal">memberships/day</span>
                    </div>
                  </div>
                  <div>
                    <div className="text-sm text-slate-400 mb-1">Days Until Opening</div>
                    <div className="text-xl font-bold text-white">
                      {daysUntilOpening != null ? daysUntilOpening : "—"}
                    </div>
                  </div>
                  <div>
                    <div className="text-sm text-slate-400 mb-1">Projected by Opening</div>
                    <div className="text-xl font-bold text-white">
                      {projectedTotal != null ? `${projectedTotal} members` : "—"}
                    </div>
                  </div>
                </div>
                {projectedTotal != null && (
                  <div className="mt-5 bg-blue-600/20 border border-blue-500/30 rounded-lg px-5 py-3">
                    <span className="text-blue-300 font-semibold">
                      At current pace, you will have {projectedTotal} members by opening day
                      {openingDate ? ` (${openingDate})` : ""}.
                    </span>
                  </div>
                )}
              </div>

              {/* Settings */}
              <div className="chart-container rounded-xl mb-8">
                <button
                  onClick={() => setSettingsOpen(!settingsOpen)}
                  className="w-full flex items-center justify-between p-5 text-left"
                >
                  <h3 className="text-lg font-semibold text-white">Settings</h3>
                  <span className="text-slate-400 text-xl">{settingsOpen ? "▲" : "▼"}</span>
                </button>
                {settingsOpen && (
                  <div className="px-6 pb-6 space-y-6">
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                      <div>
                        <label className="text-sm text-slate-400 block mb-1">Opening Date — {activeLocation}</label>
                        <input
                          type="date"
                          value={openingDate || ""}
                          onChange={(e) => updateLocationConfig("opening_date", e.target.value)}
                          className="w-full rounded-lg"
                        />
                      </div>
                      <div>
                        <label className="text-sm text-slate-400 block mb-1">Sales Start Date — {activeLocation}</label>
                        <input
                          type="date"
                          value={salesStartDate || ""}
                          onChange={(e) => updateLocationConfig("sales_start_date", e.target.value)}
                          className="w-full rounded-lg"
                        />
                      </div>
                    </div>
                  </div>
                )}
              </div>
            </>
          ) : null}
        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById("root"));
    root.render(<App />);
  </script>
</body>
</html>
```
